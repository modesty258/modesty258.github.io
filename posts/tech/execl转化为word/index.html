def center_insert_img(doc, img1, img2):
    """插入图片"""
    tables = doc.tables
    tab_2 = tables[0].rows[65].cells[30]
    ph_img2 = tab_2.paragraphs[0]
    run_img2 = ph_img2.add_run('')
    run_img2.add_picture(img2)


    tab_1 = tables[0].rows[65].cells[6]
    ph_img1 = tab_1.paragraphs[0]
    run_img1 = ph_img1.add_run('')
    run_img1.add_picture(img1)



def save_img_to_doc(img1,img2,tpl_doc,res_doc):
    """把图片保存到doc文件中的指定位置"""
    # 打开模板文件
    document = Document(tpl_doc)
    # 插入图片居中
    center_insert_img(document, img1,img2)
    # 保存结果文件
    document.save(res_doc)

import zipfile
from PIL import Image
import os
import re
def get_img_from_excel(test_file):
    new_file = test_file.replace(".xlsx",".zip")
    os.rename(test_file,new_file)
    azip = zipfile.ZipFile(new_file)
    namelist = (azip.namelist())
    print(namelist)
    img_name = 1
    for idx in range(0,len(namelist)):
        if  'media' in namelist[idx] and 'jpeg' in namelist[idx]:
            f = azip.open(namelist[idx])
            img = Image.open(f)
            img = img.convert("RGB")
            print(img_name)
            img.save('image'+str(img_name)+'.jpeg',"jpeg")
            img_name += 1
    azip.close()

def get_level(b_74_1):
    if b_74_1 == '':
        return b_74_1
    b_74_1 = b_74_1.replace('null', '2类')
    # print(b_74_1)
    b_74_1 = b_74_1.split(':')[1]
    b_74_1 = b_74_1.split(' ')[0]
    if '类类' in b_74_1:
        b_74_1 = b_74_1[:-1]
    return b_74_1

def get_next_year(b_76_1):
    if b_76_1 == '':
        return b_76_1
    b_76_1 = b_76_1.split('-')[0]
    b_76_1 = int(b_76_1) + 1
    return b_76_1
